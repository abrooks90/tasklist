CREATE DATABASE abrooks; 
USE abrooks; 
CREATE TABLE services (svcID INT NOT NULL AUTO_INCREMENT, service_description varchar(100), primary key(svcID)); 

CREATE TABLE profile (profileID INT NOT NULL AUTO_INCREMENT, netid varchar(25), fname varchar(50), lname varchar(50), email varchar(50), post_date char(10), avail_days varchar(255), avail_hours varchar(255), confirmation tinyint, primary key(profileID)); 
CREATE TABLE services_offered (svcID INT, profileID INT, foreign key(svcID) REFERENCES services(svcID), foreign key(profileID) REFERENCES profile(profileID));
